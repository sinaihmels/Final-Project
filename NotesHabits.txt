SELECT 
    user_habits.position,
    habits.name,
    habits.descriptionheader1,
    user_habits.goal, 
    habits.descriptionheader2,
    habits.icon_path,
    habits.descriptionprogress1,
    user_habits.progress, 
    habits.descriptionprogress2,
    habits.descriptionlogmore,
    habits.unit  
FROM user_habits 
JOIN habits ON user_habits.habits_id=habits.id 
WHERE user_id = 1;
